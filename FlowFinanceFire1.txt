// ====== THEME MANAGER ======
function toggleTheme(){
  document.body.classList.toggle('L');
  localStorage.setItem('ff_theme', document.body.classList.contains('L')?'L':'D');
}
if(localStorage.getItem('ff_theme')==='L') document.body.classList.add('L');

function chgPg(pgId){
  document.querySelectorAll('.pg').forEach(p=>p.classList.remove('on'));
  const target = document.getElementById('pg-'+pgId);
  if(target) target.classList.add('on');
}

// ====== AUTHENTICATION & DATABASE SERVICE (FIREBASE CORE) ======
let currentMode = 'login';
function setAuthMode(mode){
  currentMode = mode;
  document.querySelectorAll('.auth-tb').forEach(b=>b.classList.remove('on'));
  document.getElementById('tb-'+mode).classList.add('on');
  document.getElementById('auth-btn-text').innerText = mode === 'login' ? 'เข้าสู่ระบบ' : 'สมัครสมาชิก';
  document.getElementById('auth-error').innerText = '';
}

// บริการจัดการฝั่ง Auth & Database ด้วย Firebase แทนที่ Supabase
const DatabaseService = {
  // ฟังก์ชันดึงข้อมูลผู้ใช้จาก Firestore
  async getUserData(uid) {
    try {
      const docRef = window.firebaseDb.doc(window.firebaseDb, "users", uid);
      const docSnap = await window.firebaseDb.getDoc(docRef);
      if (docSnap.exists()) {
        return docSnap.data();
      } else {
        // หากไม่มีข้อมูล ให้สร้างเอกสารตั้งต้นขึ้นมา
        const defaultData = { transactions: [], budget: {}, goals: [], assets: [], debts: [] };
        await window.firebaseDb.setDoc(docRef, defaultData);
        return defaultData;
      }
    } catch (e) {
      console.error("Error getting user data:", e);
      throw e;
    }
  },

  // ฟังก์ชันบันทึก/อัปเดตข้อมูลภาพรวมไปยัง Firestore
  async saveUserData(uid, data) {
    try {
      const docRef = window.firebaseDb.doc(window.firebaseDb, "users", uid);
      await window.firebaseDb.setDoc(docRef, data, { merge: true });
    } catch (e) {
      console.error("Error saving user data:", e);
      throw e;
    }
  }
};

// จัดการเหตุการณ์เมื่อป้อนข้อมูล Auth Submit
async function handleAuthSubmit() {
  const email = document.getElementById('auth-email').value.trim();
  const password = document.getElementById('auth-password').value;
  const errDiv = document.getElementById('auth-error');
  errDiv.innerText = '';

  if (!email || !password) {
    errDiv.innerText = 'กรุณากรอกข้อมูลให้ครบถ้วน';
    return;
  }

  try {
    if (currentMode === 'login') {
      // เข้าสู่ระบบด้วย Firebase Authentication
      await window.firebaseAuth.signInWithEmailAndPassword(window.firebaseAuth, email, password);
    } else {
      // สมัครสมาชิกด้วย Firebase Authentication
      await window.firebaseAuth.createUserWithEmailAndPassword(window.firebaseAuth, email, password);
    }
  } catch (error) {
    errDiv.innerText = 'เกิดข้อผิดพลาด: ' + translateAuthError(error.code);
  }
}

function translateAuthError(code) {
  switch (code) {
    case 'auth/invalid-email': return 'รูปแบบอีเมลไม่ถูกต้อง';
    case 'auth/user-not-found': case 'auth/wrong-password': case 'auth/invalid-credential': return 'อีเมลหรือรหัสผ่านไม่ถูกต้อง';
    case 'auth/email-already-in-excess': case 'auth/email-already-in-use': return 'อีเมลนี้ถูกใช้งานแล้ว';
    case 'auth/weak-password': return 'รหัสผ่านต้องมีความยาวอย่างน้อย 6 ตัวอักษร';
    default: return 'ไม่สามารถดำเนินการได้ในขณะนี้';
  }
}

// ตรวจสอบการเปลี่ยนสถานะการเข้าสู่ระบบ (แทนที่ supabase.auth.onAuthStateChange)
function initAuthListener() {
  // รอเช็คให้ Firebase SDK ถูก Initialize เรียบร้อยก่อนทำงาน
  const checkFirebase = setInterval(() => {
    if (window.firebaseAuth) {
      clearInterval(checkFirebase);
      
      window.firebaseAuth.onAuthStateChanged(window.firebaseAuth, async (user) => {
        const authScreen = document.getElementById('auth-screen');
        const userInfo = document.getElementById('user-info');

        if (user) {
          // หาก Login อยู่ ให้ปิดหน้า Auth และดึงข้อมูลมาแสดงผล
          authScreen.classList.remove('on');
          userInfo.innerHTML = `
            <div class="nav-user-av">${user.email.substring(0,2).toUpperCase()}</div>
            <span>${user.email}</span>
            <button class="sign-out-btn" onclick="handleSignOut()">ออก</button>
          `;
          
          document.getElementById('sync-status').className = 'sync-dot ing';
          try {
            const userData = await DatabaseService.getUserData(user.uid);
            console.log("ดาวน์โหลดข้อมูลจาก Firebase สำเร็จ:", userData);
            // ที่ตรงนี้คุณสามารถสั่งให้แอป Render ชุดตัวแปรกระดานบัญชีของคุณต่อได้เลยครับ เช่น renderDash(userData);
            document.getElementById('sync-status').className = 'sync-dot ok';
          } catch (err) {
            document.getElementById('sync-status').className = 'sync-dot err';
          }

        } else {
          // หากยังไม่ได้ Login ให้แสดงหน้าจอ Auth เสมอ
          authScreen.classList.add('on');
          userInfo.innerHTML = '';
        }
      });
    }
  }, 100);
}

async function handleSignOut() {
  if (window.firebaseAuth) {
    await window.firebaseAuth.signOut(window.firebaseAuth);
  }
}

// เรียกให้ระบบเริ่มตรวจสอบ Session ตั้งแต่เปิดหน้าเว็บ
document.addEventListener("DOMContentLoaded", initAuthListener);
